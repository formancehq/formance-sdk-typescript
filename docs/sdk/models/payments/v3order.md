# V3Order

A trade order submitted to an exchange-style PSP. Orders are read-only
in the Formance API: they are fetched from the underlying connector.
Status transitions are captured via the `adjustments` array; each
adjustment is a point-in-time snapshot from the PSP.


## Example Usage

```typescript
import {
  V3Order,
  V3OrderDirectionEnum,
  V3OrderStatusEnum,
  V3OrderTypeEnum,
  V3TimeInForceEnum,
} from "@formance/formance-sdk/sdk/models/payments";

let value: V3Order = {
  v3OrderDirectionEnum: V3OrderDirectionEnum.Unknown,
  v3OrderStatusEnum: V3OrderStatusEnum.Failed,
  v3OrderTypeEnum: V3OrderTypeEnum.TakeProfitLimit,
  v3TimeInForceEnum: V3TimeInForceEnum.GoodUntilDateTime,
  baseQuantityOrdered: 720704n,
  connectorID: "<value>",
  createdAt: new Date("2024-07-01T03:08:56.264Z"),
  destinationAsset: "<value>",
  id: "<id>",
  provider: "<value>",
  reference: "<value>",
  sourceAsset: "<value>",
  updatedAt: new Date("2026-07-09T01:50:01.037Z"),
};
```
